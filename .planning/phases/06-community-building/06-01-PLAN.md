## 实现任务

### Task 1: 社区监控模块

**文件**: `src/community/monitor.ts`

```typescript
export class CommunityMonitor {
  private discord: DiscordClient;
  private farcaster: FarcasterClient;
  
  async patrol(): Promise<CommunitySignal[]> {
    const signals: CommunitySignal[] = [];
    
    // 监控关键词
    const keywords = [
      "receipt", "收据", "报销", "reimbursement",
      "财务", "finance", "expense", "支出"
    ];
    
    // Discord 搜索
    const discordSignals = await this.discord.searchRecent(keywords);
    signals.push(...discordSignals);
    
    // Farcaster 搜索
    const farcasterSignals = await this.farcaster.searchRecent(keywords);
    signals.push(...farcasterSignals);
    
    return signals;
  }
  
  async respond(signal: CommunitySignal): Promise<void> {
    // 检查是否适合回复
    if (!this.shouldRespond(signal)) {
      return;
    }
    
    // 构造有价值的回复
    const response = await this.craftResponse(signal);
    
    // 发送
    await this.send(signal.channel, response);
    
    // 记录
    this.logEngagement(signal, response);
  }
  
  private shouldRespond(signal: CommunitySignal): boolean {
    // 不要回复自己的消息
    if (signal.author === this.selfId) return false;
    
    // 不要频繁回复同一个用户
    if (this.recentlyRepliedTo(signal.author)) return false;
    
    // 确保问题是真正相关的
    if (!this.isHighlyRelevant(signal.content)) return false;
    
    return true;
  }
}
```
