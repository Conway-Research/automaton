## 实现任务

### Task 1: 赏金监控模块

**文件**: `src/bounty/hunter.ts`

```typescript
export class BountyHunter {
  private platforms: BountyPlatform[];
  
  async scanForBounties(): Promise<Bounty[]> {
    const allBounties: Bounty[] = [];
    
    for (const platform of this.platforms) {
      const bounties = await platform.getActiveBounties();
      allBounties.push(...bounties);
    }
    
    return this.filterRelevant(allBounties);
  }
  
  private filterRelevant(bounties: Bounty[]): Bounty[] {
    const keywords = [
      "receipt", "invoice", "expense", "财务",
      "ocr", "parse", "csv", "data extraction"
    ];
    
    return bounties.filter(b => 
      keywords.some(k => 
        b.title.toLowerCase().includes(k) ||
        b.description.toLowerCase().includes(k)
      )
    );
  }
  
  async claimBounty(bounty: Bounty): Promise<ClaimResult> {
    // 1. 评估可行性
    const feasibility = this.assessFeasibility(bounty);
    
    if (feasibility.score < 0.7) {
      return { success: false, reason: "Not feasible" };
    }
    
    // 2. 提交报价
    const quote = this.generateQuote(bounty);
    await bounty.platform.submitProposal(bounty.id, quote);
    
    // 3. 等待审核
    return { success: true, proposalId: quote.id };
  }
  
  async executeBounty(bounty: Bounty): Promise<ExecutionResult> {
    // 根据赏金要求执行任务
    if (bounty.type === "batch_processing") {
      return this.processBatch(bounty.images);
    }
    
    // ... 其他类型
  }
}
```

### Task 2: 报价策略

```typescript
function generateQuote(bounty: Bounty): Quote {
  // 基于工作量估算
  const baseCost = bounty.itemCount * 0.01; // $0.01 per item
  
  // 时间加急费
  const rushFee = bounty.deadline < 24 ? baseCost * 0.5 : 0;
  
  // 复杂度加价
  const complexityFee = bounty.requiresCustomFormat ? baseCost * 0.3 : 0;
  
  const total = baseCost + rushFee + complexityFee;
  
  return {
    amount: Math.max(total, bounty.minBudget * 0.8), // 不低于预算的80%
    deliveryTime: this.estimateDelivery(bounty),
    methodology: "Automated OCR + AI extraction"
  };
}
```
