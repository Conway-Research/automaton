## 执行策略

### Step 1: 扫描 DAO 金库地址

```typescript
async function scanDAOTreasuries(): Promise<DAOTarget[]> {
  // 已知的大型 DAO 金库地址
  const knownDAOs = [
    { name: "Nouns DAO", address: "0x..." },
    { name: "Uniswap DAO", address: "0x..." },
    // ...
  ];
  
  const targets = [];
  
  for (const dao of knownDAOs) {
    // 检查该地址的转账频率
    const transfers = await getRecentTransfers(dao.address, 30); // 30天
    
    // 筛选有频繁报销转账的 DAO
    if (transfers.filter(t => isReimbursement(t)).length > 10) {
      targets.push({
        ...dao,
        reimburseCount: transfers.length,
        potential: "HIGH"
      });
    }
  }
  
  return targets;
}
```
